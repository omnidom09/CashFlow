const API_URL = "https://cashflow-lnaf.onrender.com"; async function trackOrder() { const orderId = document.getElementById("orderId").value; if (!orderId) { alert("Enter order ID"); return; } try { const response = await fetch(`${API_URL}/track/${orderId}`); const data = await response.json(); document.getElementById("result").innerHTML = "Status: " + data.status; } catch (error) { document.getElementById("result").innerHTML = "Error connecting to server"; } } async function makePayment() { try { const response = await fetch(`${API_URL}/payment`, { method: "POST" }); const data = await response.json(); document.getElementById("paymentResult").innerHTML = "Payment ID: " + data.payment_id; } catch (error) { document.getElementById("paymentResult").innerHTML = "Payment failed"; } }
